from ipyleaflet import Map, Marker, basemaps, basemap_to_tiles
m = Map(
  basemap=basemap_to_tiles(
    basemaps.USGS.USImageryTopo, "2020-09-31"
  ),
  center=(39.712950, -105.406600),
  zoom=15
)
m.add_layer(Marker(location=(39.712950, -105.406600)))
